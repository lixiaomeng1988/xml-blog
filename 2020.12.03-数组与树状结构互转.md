### 一维数组转换为树状结构

```js
const data = [
    { id: '0', name: '0' },
    { id: '0-0', parentId: '0', name: '0-0' },
    { id: '0-1', parentId: '0', name: '0-1' },

    { id: '1', name: '1' },
    { id: '1-0', parentId: '1', name: '1-0' },
];

function toTree(arr, deep) {
    const right = [];

    const loop = (r, parentId, d) => {
      let left = [];

      if (d <= 0) return [];

      if (!parentId) {
        left = r.filter(item => {
          if (item.parentId) {
            right[right.length] = item;
          }

          return !item.parentId;
        });
      } else {
        left = r.filter((item, index) => {
          if (item.parentId === parentId) {
            right.slice(index, 1);
            return true;
          }
          return false;
        });
      }

      return left.map(item => {
        return {
          ...item,
          children: loop(right, item.id, d - 1),
        };
      });
    };

    return loop(arr, null, deep);
  }

console.log(toTree(data));
```

```js
const data = [
    { id: '0', name: '0' },
    { id: '0-0', parentId: '0', name: '0-0' },
    { id: '0-1', parentId: '0', name: '0-1' },

    { id: '1', name: '1' },
    { id: '1-0', parentId: '1', name: '1-0' },
  ];

function toTree(list) {
  const map = {};

  list.forEach(item => {
    map[item.id] = item;
  });

  return list.filter(item => {
    const parent = map[item.parentId];

    if (parent) {
      parent.children = parent.children || [];
      map[item.parentId].children.push(item);
    };

    return !parent;
  });
}

console.log(toTree(data));
```
